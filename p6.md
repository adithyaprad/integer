//index.js


// 1. Object literal
const obj1 = { name: "John", age: 25 };

// 2. Constructor function
function Person(name, age) {
  this.name = name;
  this.age = age;
}
const obj2 = new Person("John", 25);

// 3. Object.create()
const proto = {name: "John", age: 12}
const obj3 = Object.create(proto);

// 4. Class
class Person {
  constructor(name) { this.name = name; }
}
const obj4 = new Person("Rahul");

// Iterate using for...in
for (let key in obj1) {
  console.log(key + " : " + obj1[key]);
}
